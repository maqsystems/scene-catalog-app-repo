#include <tunables/global>

profile scene_catalog flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6 overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # App data/config
  /data/** rw,
  /share/** rw,
  /opt/scene_catalog/** r,

  # Python runtime
  /usr/bin/python3 ix,

  # Start app profile
  /usr/bin/python3 cx -> scene_catalog_python,

  profile scene_catalog_python flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    signal (receive) peer=scene_catalog,

    /opt/scene_catalog/** r,
    /usr/bin/python3 rix,
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/passwd r,
    /tmp/** rwk,
    /data/** rw,
    /share/** rw,

    # HTTPS and network access to supervisor/core API
    network inet stream,
    network inet6 stream,
  }
}
